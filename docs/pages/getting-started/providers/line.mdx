### 配置

<Code>
  <Code.Next>

```ts filename="/auth.ts"
import NextAuth from "next-auth"
import Line from "next-auth/providers/line"

export const { handlers, auth, signIn, signOut } = NextAuth({
  providers: [Line],
})
```

  </Code.Next>
  <Code.Qwik>
  
```ts filename="/src/routes/plugin@auth.ts"
import { QwikAuth$ } from "@auth/qwik"
import Line from "@auth/qwik/providers/line"

export const { onRequest, useSession, useSignIn, useSignOut } = QwikAuth$(
  () => ({
    providers: [Line],
  })
)
```

  </Code.Qwik>
  <Code.Svelte>

```ts filename="/src/auth.ts"
import { SvelteKitAuth } from "@auth/sveltekit"
import Line from "@auth/sveltekit/providers/line"

export const { handle, signIn, signOut } = SvelteKitAuth({
  providers: [Line],
})
```

  </Code.Svelte>
  <Code.Express>

```ts filename="/src/app.ts"
import { ExpressAuth } from "@auth/express"
import Line from "@auth/express/providers/line"

app.use("/auth/*", ExpressAuth({ providers: [Line] }))
```

  </Code.Express>
</Code>

在 https://developers.line.biz/console/ 创建提供商和 LINE 登录渠道。在 LINE Login 下的渠道设置中，启用 web 应用并按上述定义配置回调 URL。

### 注意事项

- 如需获取邮箱地址，需申请邮箱地址权限。打开 [Line 开发者控制台](https://developers.line.biz/console/)，进入您的登录渠道。滚动至底部找到 **OpenID Connect** -> **Email address permission**。点击 **Apply** 并按照指引操作。
