```
AUTH_KEYCLOAK_ID
```

### 配置

<Code>
  <Code.Next>

```ts filename="/auth.ts"
import NextAuth from "next-auth"
import Keycloak from "next-auth/providers/keycloak"

export const { handlers, auth, signIn, signOut } = NextAuth({
  providers: [Keycloak],
})
```

  </Code.Next>
  <Code.Qwik>
  
```ts filename="/src/routes/plugin@auth.ts"
import { QwikAuth$ } from "@auth/qwik"
import Keycloak from "@auth/qwik/providers/keycloak"

export const { onRequest, useSession, useSignIn, useSignOut } = QwikAuth$(
  () => ({
    providers: [Keycloak],
  })
)
```

  </Code.Qwik>
  <Code.Svelte>

```ts filename="/src/auth.ts"
import { SvelteKitAuth } from "@auth/sveltekit"
import Keycloak from "@auth/sveltekit/providers/keycloak"

export const { handle, signIn, signOut } = SvelteKitAuth({
  providers: [Keycloak],
})
```

  </Code.Svelte>
  <Code.Express>

```ts filename="/src/app.ts"
import { ExpressAuth } from "@auth/express"
import Keycloak from "@auth/express/providers/keycloak"

app.use("/auth/*", ExpressAuth({ providers: [Keycloak] }))
```

  </Code.Express>
</Code>

启用"客户端认证"选项以在凭证选项卡中获取您的客户端密钥。

在 v20 版本之前，需在 Keycloak 中创建一个访问类型为"confidential"的`openid-connect`客户端。

- Issuer 应包含 realm 名称 - 例如 `https://my-keycloak-domain.com/realms/My_Realm`
